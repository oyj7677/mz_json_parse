# String Resource Explorer Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a fourth MZ Tools hub tool that lets users upload multilingual Excel string resource workbooks, search by visible string content, and inspect Android qualifier-based language values per resource ID.

**Architecture:** Add focused string-resource parsing modules beside the existing browser app, then wire a new tool view into the current static `index.html`/`app.js`/`styles.css` pattern. Workbook files are parsed in the browser using a vendored SheetJS browser bundle, then normalized into resource rows that the app can search and render without server upload.

**Tech Stack:** Static HTML/CSS/ES modules, Node test runner, SheetJS `xlsx` browser bundle, Vercel static hosting.

---

## File Structure

- Create `web/public/string-resource-core.js`
  - Pure helper module for language qualifier normalization, sheet detection, row normalization, search filtering, visible qualifier ordering, and display values.
- Create `web/public/string-resource-xlsx.js`
  - Browser adapter that loads the vendored SheetJS bundle and converts uploaded `.xlsx` files into the normalized workbook shape consumed by `string-resource-core.js`.

```js
const ID_COLUMN_NAMES = [
  'MOBIS LID',
  'HMC UID',
  'String ID',
  'PromptLID',
  'CheckLID',
  'ID'
];

const QUALIFIER_ALIASES = new Map([
  ['korean', 'ko'],
  ['values-ko', 'ko'],
  ['english us', 'en-rUS'],
  ['english usa', 'en-rUS'],
  ['english (us)', 'en-rUS'],
  ['values-en-rus', 'en-rUS'],
  ['english uk', 'en-rGB'],
  ['english gb', 'en-rGB'],
  ['english (uk)', 'en-rGB'],
  ['values-en-rgb', 'en-rGB'],
  ['english au', 'en-rAU'],
  ['english (australia)', 'en-rAU'],
  ['values-en-rau', 'en-rAU'],
  ['spanish (mexico)', 'es-rMX'],
  ['spanish mexico', 'es-rMX'],
  ['values-es-rmx', 'es-rMX'],
  ['spanish (spain)', 'es-rES'],
  ['spanish spain', 'es-rES'],
  ['values-es-res', 'es-rES'],
  ['french (canada)', 'fr-rCA'],
  ['french canada', 'fr-rCA'],
  ['values-fr-rca', 'fr-rCA'],
  ['portuguese (brazil)', 'pt-rBR'],
  ['portuguese brazil', 'pt-rBR'],
  ['values-pt-rbr', 'pt-rBR'],
  ['chinese (simplified, china)', 'zh-rCN'],
  ['chinese simplified china', 'zh-rCN'],
  ['values-zh-rcn', 'zh-rCN']
]);

export const STRING_RESOURCE_DEFAULT_QUALIFIERS = [
  'ko',
  'en-rUS',
  'en-rGB',
  'en-rAU',
  'es-rMX',
  'es-rES',
  'fr-rCA',
  'pt-rBR',
  'zh-rCN'
];

export function normalizeStringResourceQualifier(header) {
  const value = String(header ?? '').trim();
  if (!value) {
    return '';
  }

  const lowered = value.toLowerCase().replace(/\s+/g, ' ');
  if (QUALIFIER_ALIASES.has(lowered)) {
    return QUALIFIER_ALIASES.get(lowered);
  }

  const valuesMatch = lowered.match(/^values-([a-z]{2})(?:-r([a-z]{2}))?$/i);
  if (valuesMatch) {
    const language = valuesMatch[1].toLowerCase();
    const region = valuesMatch[2] ? `-r${valuesMatch[2].toUpperCase()}` : '';
    return `${language}${region}`;
  }

  return '';
}

export function detectStringResourceSheet(sheet) {
  const rows = Array.isArray(sheet?.rows) ? sheet.rows : [];
  const columnNames = collectColumnNames(rows);
  const idColumns = columnNames.filter((name) => ID_COLUMN_NAMES.includes(name));
  const languageColumns = columnNames
    .map((name) => ({ name, qualifier: normalizeStringResourceQualifier(name) }))
    .filter((column) => column.qualifier);

  return {
    idColumns,
    isCandidate: idColumns.length > 0 && languageColumns.length > 0,
    languageColumns,
    name: String(sheet?.name ?? ''),
    rowCount: Math.max(0, rows.length - countHeaderLikeRows(rows))
  };
}

export function normalizeStringResourceWorkbook(workbook, fileName = workbook?.source ?? '') {
  const sheets = Array.isArray(workbook?.sheets) ? workbook.sheets : [];
  const sheetSummaries = sheets.map((sheet) => detectStringResourceSheet(sheet));
  const rows = [];

  for (const sheet of sheets) {
    const sheetName = String(sheet?.name ?? '').trim();
    const summary = detectStringResourceSheet(sheet);
    if (!summary.isCandidate) {
      continue;
    }

    const sourceRows = Array.isArray(sheet?.rows) ? sheet.rows : [];
    for (const sourceRow of sourceRows) {
      const values = normalizeValues(sourceRow?.values);
      if (isHeaderRow(values, summary)) {
        continue;
      }

      const languages = {};
      const duplicateLanguages = {};
      for (const column of summary.languageColumns) {
        const text = scalarString(values[column.name]);
        if (!text) {
          continue;
        }
        if (languages[column.qualifier] === undefined) {
          languages[column.qualifier] = text;
        } else {
          duplicateLanguages[column.qualifier] ??= [languages[column.qualifier]];
          duplicateLanguages[column.qualifier].push(text);
        }
      }

      if (Object.keys(languages).length === 0) {
        continue;
      }

      const idFields = {};
      for (const columnName of summary.idColumns) {
        const text = scalarString(values[columnName]);
        if (text) {
          idFields[columnName] = text;
        }
      }

      const rowNumber = Number(sourceRow?.rowNumber ?? rows.length + 1);
      const resourceId = Object.values(idFields).find(Boolean) || `${sheetName}:${rowNumber}`;
      rows.push({
        duplicateLanguages,
        fileName: String(fileName || workbook?.source || 'uploaded.xlsx'),
        id: `${String(fileName || workbook?.source || 'uploaded.xlsx')}:${sheetName}:${rowNumber}`,
        idFields,
        languages,
        metadata: pickMetadata(values, summary),
        originalValues: values,
        resourceId,
        rowNumber,
        sheetName
      });
    }
  }

  return {
    fileName: String(fileName || workbook?.source || 'uploaded.xlsx'),
    rows,
    sheetSummaries
  };
}

export function resolveStringResourceQualifiers(rows) {
  const found = new Set();
  for (const row of rows) {
    for (const qualifier of Object.keys(row?.languages ?? {})) {
      found.add(qualifier);
    }
  }

  const ordered = STRING_RESOURCE_DEFAULT_QUALIFIERS.filter((qualifier) => found.has(qualifier));
  const extras = [...found]
    .filter((qualifier) => !STRING_RESOURCE_DEFAULT_QUALIFIERS.includes(qualifier))
    .sort((a, b) => a.localeCompare(b));
  return [...ordered, ...extras];
}

export function filterStringResourceRows(rows, query) {
  const groups = parseSearchGroups(query);
  if (groups.length === 0) {
    return [];
  }

  return rows.filter((row) => groups.every((alternatives) => alternatives.some((term) => termMatchesRow(row, term))));
}

function collectColumnNames(rows) {
  const names = new Set();
  for (const row of rows) {
    const values = normalizeValues(row?.values);
    for (const key of Object.keys(values)) {
      if (String(key).trim()) {
        names.add(String(key).trim());
      }
    }
  }
  return [...names];
}

function normalizeValues(values) {
  if (!values || typeof values !== 'object') {
    return {};
  }
  return Object.fromEntries(Object.entries(values).map(([key, value]) => [String(key).trim(), value]));
}

function scalarString(value) {
  if (value === null || value === undefined) {
    return '';
  }
  if (typeof value === 'string' || typeof value === 'number' || typeof value === 'boolean') {
    return String(value).trim();
  }
  return '';
}

function countHeaderLikeRows(rows) {
  return rows.filter((row) => {
    const values = normalizeValues(row?.values);
    return Object.entries(values).some(([key, value]) => scalarString(value) === key);
  }).length;
}

function isHeaderRow(values, summary) {
  const checkedColumns = [...summary.idColumns, ...summary.languageColumns.map((column) => column.name)];
  return checkedColumns.some((columnName) => scalarString(values[columnName]) === columnName);
}

function pickMetadata(values, summary) {
  const ignored = new Set([
    ...summary.idColumns,
    ...summary.languageColumns.map((column) => column.name)
  ]);
  const metadata = {};
  for (const [key, value] of Object.entries(values)) {
    const text = scalarString(value);
    if (text && !ignored.has(key)) {
      metadata[key] = text;
    }
  }
  return metadata;
}

function parseSearchGroups(query) {
  return splitSearch(query, ',')
    .map((part) => splitSearch(part, '|').filter(Boolean))
    .filter((group) => group.length > 0);
}

function splitSearch(query, delimiter) {
  return String(query ?? '')
    .split(delimiter)
    .map((part) => part.trim())
    .filter(Boolean);
}

function termMatchesRow(row, term) {
  const values = [
    row.resourceId,
    ...Object.values(row.idFields ?? {}),
    ...Object.values(row.languages ?? {}),
    ...Object.values(row.metadata ?? {}),
    row.fileName,
    row.sheetName
  ].map((value) => String(value ?? '').trim()).filter(Boolean);

  if (term.includes('*')) {
    const regex = new RegExp(`^${term.split('*').map(escapeRegExp).join('.*')}$`, 'i');
    return values.some((value) => regex.test(value));
  }

  const needle = term.toLowerCase();
  return values.some((value) => value.toLowerCase().includes(needle));
}

function escapeRegExp(value) {
  return String(value).replace(/[.*+?^${}()|[\]\\]/g, '\\$&');
}
```

```js
export async function parseStringResourceWorkbookFile(file, root = globalThis) {
  const xlsx = getBrowserXlsx(root);
  const buffer = await file.arrayBuffer();
  const workbook = xlsx.read(buffer, { type: 'array' });
  return convertSheetJsonToWorkbook({
    SheetNames: workbook.SheetNames,
    Sheets: Object.fromEntries(workbook.SheetNames.map((name) => [
      name,
      xlsx.utils.sheet_to_json(workbook.Sheets[name], { header: 1, blankrows: false, defval: '' })
    ]))
  }, file.name);
}

export function convertSheetJsonToWorkbook(sheetJsonWorkbook, source) {
  const sheetNames = Array.isArray(sheetJsonWorkbook?.SheetNames) ? sheetJsonWorkbook.SheetNames : [];
  return {
    source,
    sheets: sheetNames.map((name) => ({
      name,
      rows: rowsToObjects(sheetJsonWorkbook.Sheets?.[name] ?? [])
    }))
  };
}

function rowsToObjects(rows) {
  if (!Array.isArray(rows) || rows.length === 0) {
    return [];
  }

  const header = rows[0].map((value, index) => {
    const text = String(value ?? '').trim();
    return text || `Column ${index + 1}`;
  });

  return rows.map((row, rowIndex) => ({
    rowNumber: rowIndex + 1,
    values: Object.fromEntries(header.map((columnName, columnIndex) => [
      columnName,
      row?.[columnIndex] ?? ''
    ]))
  }));
}
```

```js
async function registerStringResourceFiles(fileList) {
  const files = Array.from(fileList ?? []);
  if (files.length === 0) {
    setStringResourceUploadStatus('선택된 엑셀 파일이 없습니다.');
    return;
  }

  let addedCount = 0;
  let errorCount = 0;

  for (const file of files) {
    try {
      const workbook = await parseStringResourceWorkbookFile(file);
      const normalized = normalizeStringResourceWorkbook(workbook, file.name);
      state.stringResource.files.push({
        fileName: file.name,
        rows: normalized.rows,
        sheetSummaries: normalized.sheetSummaries
      });

      for (const summary of normalized.sheetSummaries) {
        if (summary.isCandidate) {
          state.stringResource.selectedSheetIds.add(stringResourceSheetId(file.name, summary.name));
        }
      }

      addedCount += 1;
    } catch (error) {
      errorCount += 1;
      state.stringResource.errors.push(`${file.name}: ${error instanceof Error ? error.message : String(error)}`);
    }
  }

  state.stringResource.rows = state.stringResource.files.flatMap((file) => file.rows);
  setStringResourceUploadStatus(`${addedCount}개 파일 등록, 오류 ${errorCount}개`);
  renderStringResource();
}
```

```js
function openStringResourceDetail(rowId) {
  state.stringResource.modalRowId = rowId;
  renderStringResourceDetail();
  elements.stringResourceDetailModal.hidden = false;
  elements.closeStringResourceDetailButton.focus();
}

function closeStringResourceDetail() {
  state.stringResource.modalRowId = '';
  elements.stringResourceDetailModal.hidden = true;
}

function renderStringResourceDetail() {
  const row = state.stringResource.rows.find((item) => item.id === state.stringResource.modalRowId);
  elements.stringResourceDetailBody.replaceChildren();

  if (!row) {
    elements.stringResourceDetailTitle.textContent = 'String Resource 상세';
    elements.stringResourceDetailMeta.textContent = '선택된 리소스가 없습니다.';
    return;
  }

  elements.stringResourceDetailTitle.textContent = row.resourceId;
  elements.stringResourceDetailMeta.textContent = `${row.fileName} · ${row.sheetName} · row ${row.rowNumber}`;

  const grid = document.createElement('div');
  grid.className = 'string-resource-detail-grid';

  const languageSection = document.createElement('section');
  languageSection.innerHTML = '<h3>언어별 문자열</h3>';
  for (const qualifier of resolveStringResourceQualifiers([row])) {
    const value = row.languages[qualifier] ?? '';
    if (!value) {
      continue;
    }
    const valueRow = document.createElement('div');
    valueRow.className = 'string-resource-value-row';
    const key = document.createElement('strong');
    key.textContent = qualifier;
    const text = document.createElement('span');
    text.textContent = value;
    const copy = document.createElement('button');
    copy.className = 'ghost-button';
    copy.type = 'button';
    copy.textContent = 'Copy';
    copy.addEventListener('click', () => copyStringResourceValue(value));
    valueRow.append(key, text, copy);
    languageSection.append(valueRow);
  }

  const metaSection = document.createElement('section');
  metaSection.innerHTML = '<h3>출처와 메타</h3>';
  metaSection.append(
    renderStringResourceKeyValueList({
      File: row.fileName,
      Sheet: row.sheetName,
      Row: row.rowNumber,
      ...row.idFields,
      ...row.metadata
    })
  );

  const rawSection = document.createElement('section');
  rawSection.innerHTML = '<h3>원본 행</h3>';
  rawSection.append(renderStringResourceKeyValueList(row.originalValues));

  grid.append(languageSection, metaSection, rawSection);
  elements.stringResourceDetailBody.append(grid);
}

function renderStringResourceKeyValueList(values) {
  const dl = document.createElement('dl');
  dl.className = 'string-resource-key-values';
  for (const [key, value] of Object.entries(values ?? {})) {
    if (String(value ?? '').trim() === '') {
      continue;
    }
    const dt = document.createElement('dt');
    dt.textContent = key;
    const dd = document.createElement('dd');
    dd.textContent = String(value);
    dl.append(dt, dd);
  }
  return dl;
}

async function copyStringResourceValue(value) {
  try {
    await navigator.clipboard.writeText(String(value ?? ''));
  } catch {
    window.prompt('복사할 값을 선택하세요.', String(value ?? ''));
  }
}
```
